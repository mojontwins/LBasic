649pixelfont_t* internals_build_font( uint32_t* font ) {
650 uint32_t const* data = font;
651 int chr_width = (int)*data++;
652 int chr_height = (int)*data++;
653 int chr_baseline = (int)*data++;
654 int chr_mod = 256 / chr_width;
655 PIXELFONT_U8 pixels[ 32 * 32 ];
656 pixelfont_builder_t* builder = pixelfont_builder_create( chr_height, chr_baseline, chr_height, NULL );
657 for( int c = 0; c < 256; ++c ) {
658 int sx = ( c % chr_mod ) * chr_width;
659 int sy = ( c / chr_mod ) * chr_height;
660 for( int y = 0; y < chr_height; ++y ) {
661 for( int x = 0; x < chr_width; ++x ) {
662 int v = ( sx + x ) / 32;
663 int u = ( sx + x ) - ( v * 32 );
664 uint32_t b = data[ v + ( sy + y ) * 8 ];
665 if( b & ( 1 << u ) ) {
666 pixels[ x + y * chr_width ] = 1;
667 } else {
668 pixels[ x + y * chr_width ] = 0;
669 }
670 }
671 }
672 pixelfont_builder_glyph( builder, c, chr_width, pixels, chr_width, 0, chr_width );
673 }
674 pixelfont_t* pixelfont = pixelfont_builder_font( builder );
675 pixelfont_t* output = (pixelfont_t*) malloc( pixelfont->size_in_bytes );
676 memcpy( output, pixelfont, pixelfont->size_in_bytes );
677 pixelfont_builder_destroy( builder );
678 return output;
679}

1250static bool blitclip( int* px, int *py, int width, int height, int* psrcx, int* psrcy, int* psrcw, int* psrch ) {
1251 int x = *px;
1252 int y = *py;
1253 int srcx = *psrcx;
1254 int srcy = *psrcy;
1255 int srcw = *psrcw;
1256 int srch = *psrch;
1257 if( x < 0 ) {
1258 srcx -= x;
1259 srcw += x;
1260 x = 0;
1261 }
1262 if( y < 0 ) {
1263 srcy -= y;
1264 srch += y;
1265 y = 0;
1266 }
1267 if( srcx < 0 ) {
1268 x += srcx;
1269 srcw += srcx;
1270 srcx = 0;
1271 }
1272 if( srcy < 0 ) {
1273 y += srcy;
1274 srch += srcy;
1275 srcy = 0;
1276 }
1277 if( srcx + srcw >= width ) {
1278 srcw += ( width - ( srcx + srcw ) );
1279 }
1280 if( srcy + srch >= height ) {
1281 srch += ( height - ( srcy + srch ) );
1282 }
1283 if( x + srcw >= internals->draw.width ) {
1284 srcw += ( internals->draw.width - ( x + srcw ) );
1285 }
1286 if( y + srch >= internals->draw.height ) {
1287 srch += ( internals->draw.height - ( y + srch ) );
1288 }
1289 if( srcw <= 0 || srch <= 0 || x + srcw < 0 || y + srch < 0 || x > internals->draw.width || y > internals->draw.height ) {
1290 return false;
1291 } else {
1292 *px = x;
1293 *py = y;
1294 *psrcx = srcx;
1295 *psrcy = srcy;
1296 *psrcw = srcw;
1297 *psrch = srch;

1299 }

1348static void load_gif_frame( void* data, struct GIF_WHDR* whdr ) {
1349 struct gif_load_context_t* context = (struct gif_load_context_t*) data;
1350 if( context->pixels ) {
1351 return;
1352 }
1353 context->width = whdr->xdim;
1354 context->height = whdr->ydim;
1355 context->palcount = whdr->clrs;
1356 memcpy( context->palette, whdr->cpal, whdr->clrs * 3 );
1357 context->pixels = (uint8_t*)malloc( whdr->xdim * whdr->ydim );
1358 memset( context->pixels, 0, whdr->xdim * whdr->ydim );
1359
1360 uint32_t ddst = (uint32_t)( whdr->xdim * whdr->fryo + whdr->frxo );
1361 uint32_t iter = ( whdr->intr ) ? 0 : 4;
1362 uint32_t ifin = ( !iter ) ? 4 : 5; // interlacing support
1363 for( uint32_t dsrc = (uint32_t)-1; iter < ifin; ++iter ) {
1364 for( uint32_t yoff = 16U >> ((iter > 1)? iter : 1), y = (8 >> iter) & 7;
1365 y < (uint32_t)whdr->fryd; y += yoff) {
1366 for( uint32_t x = 0; x < (uint32_t)whdr->frxd; x++) {
1367 if (whdr->tran != (long)whdr->bptr[++dsrc]) {
1368 context->pixels[(uint32_t)whdr->xdim * y + x + ddst] = whdr->bptr[dsrc];
1369 }
1370 }
1371 }
1372 }

1682void fillpoly( int* points_xy, int count ) {
1683 #define MAX_POLYGON_POINTS 256
1684 static int node_x[ MAX_POLYGON_POINTS ];
1685
1686 if( internals->screen.font ) return;
1687
1688 if( count <= 0 || count > MAX_POLYGON_POINTS ) {
1689 return;
1690 }
1691
1692 int color = internals->graphics.color;
1693
1694 int min_y = points_xy[ 0 + 1 ];
1695 int max_y = min_y;
1696
1697 // find extents
1698 for( int i = 1; i < count; ++i ) {
1699 if( points_xy[ i * 2 + 1 ] < min_y ) min_y = points_xy[ i * 2 + 1 ];
1700 if( points_xy[ i * 2 + 1 ] > max_y ) max_y = points_xy[ i * 2 + 1 ];
1701 }
1702
1703 for( int y = min_y; y < max_y; ++y ) {
1704 // find intersection points_xy
1705 int nodes = 0;
1706 int j = count - 1;
1707 for( int i = 0; i < count; ++i ) {
1708 if( ( points_xy[ i * 2 + 1 ] <= y && points_xy[ j * 2 + 1 ] > y ) ||
1709 ( points_xy[ j * 2 + 1 ] <= y && points_xy[ i * 2 + 1 ] > y ) ) {
1710
1711 int dx = points_xy[ j * 2 + 0 ] - points_xy[ i * 2 + 0 ];
1712 int dy = points_xy[ j * 2 + 1 ] - points_xy[ i * 2 + 1 ];
1713 node_x[ nodes++ ] = points_xy[ i * 2 + 0 ] + ( ( y - points_xy[ i * 2 + 1 ] ) * dx ) / dy ;
1714 }
1715 j = i;
1716 }
1717
1718 // sort by x
1719 int xi = 0;
1720 while( xi < nodes - 1 ) {
1721 if( node_x[ xi ] > node_x[ xi + 1 ] ) {
1722 int swap = node_x[ xi ];
1723 node_x[ xi ] = node_x[ xi + 1 ];
1724 node_x[ xi + 1 ] = swap;
1725 if( xi ) --xi;
1726 } else {
1727 ++xi;
1728 }
1729 }
1730
1731 for( int i = 0; i < nodes; i += 2 ) {
1732 hline( node_x[ i ], y, node_x[ i + 1 ] - node_x[ i ], color );
1733 }
1734 }

1765void floodfill( int x, int y ) {
1766 #define FILLMAX 10000 /* FILLMAX depth of stack */
1767
1768 #define FILLPUSH( Y, XL, XR, DY ) /* FILLPUSH new segment on stack */ \
1769 if( sp < stack + FILLMAX && Y + ( DY ) >= 0 && Y + ( DY ) < internals->draw.height ) \
1770 { sp->y = Y; sp->xl = XL; sp->xr = XR; sp->dy = DY; ++sp; }
1771
1772 #define FILLPOP( Y, XL, XR, DY ) /* FILLPOP segment off stack */ \
1773 { --sp; Y = sp->y + ( DY = sp->dy ); XL = sp->xl; XR = sp->xr; }
1774
1775 if( internals->screen.font ) return;
1776 int color = internals->graphics.color;
1777
1778 /*
1779 * Filled horizontal segment of scanline y for xl<=x<=xr.
1780 * Parent segment was on line y-dy. dy=1 or -1
1781 */
1782 struct segment_t { int y, xl, xr, dy; };
1783
1784 int l, x1, x2, dy;
1785 int ov; /* old pixel value */
1786 struct segment_t stack[ FILLMAX ], *sp = stack; /* stack of filled segments */
1787
1788 ov = getpixel( x, y ); /* read pv at seed point */
1789 if( ov == color || x < 0 || x >= internals->draw.width || y < 0 || y >= internals->draw.height ) return;
1790 FILLPUSH( y, x, x, 1 ); /* needed in some cases */
1791 FILLPUSH( y + 1, x, x, -1 ); /* seed segment (FILLPOPped 1st) */
1792
1793 while( sp > stack ) {
1794 /* FILLPOP segment off stack and fill a neighboring scan line */
1795 FILLPOP( y, x1, x2, dy );
1796 /*
1797 * segment of scan line y-dy for x1<=x<=x2 was previously filled,
1798 * now explore adjacent pixels in scan line y
1799 */
1800 int xs;
1801 for( x = x1; x >= 0 && getpixel( x, y ) == ov; --x ) /* nothing */;
1802 hline( x + 1, y, x1 - x, color );
1803 if( x >= x1 ) goto skip;
1804 l = x + 1;
1805 if( l < x1 ) FILLPUSH( y, l, x1 - 1, -dy ); /* leak on left? */
1806 x = x1 + 1;
1807 do {
1808 xs = x;
1809 for( ; x < internals->draw.width && getpixel( x, y ) == ov; ++x ) /* nothing */;
1810 hline( xs, y, x - xs, color );
1811 FILLPUSH( y, l, x - 1, dy);
1812 if( x > x2 + 1 ) FILLPUSH( y, x2 + 1, x - 1, -dy); /* leak on right? */
1813 skip:
1814 for( x++; x <= x2 && getpixel( x, y ) != ov; ++x );
1815 l = x;
1816 } while( x <= x2 );
1817 }
1818
1819 #undef FILLMAX
1820}
